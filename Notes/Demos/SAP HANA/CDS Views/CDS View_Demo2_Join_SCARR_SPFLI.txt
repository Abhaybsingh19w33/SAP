define view ZVK_CDS_SCARR_SPFLI as 
select from scarr SC
left outer join spfli SP
on SC.carrid = SP.carrid 
{
   SC.carrid,
   SC.carrname,  
   SP.countryfr,
   SP.cityfrom
  
};

Note: You can use Right Outer join also in the  above syntax.
