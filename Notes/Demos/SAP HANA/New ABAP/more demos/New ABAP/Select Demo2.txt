*Old Syntax
REPORT Z.
DATA WA TYPE SPFLI.
SELECT SINGLE *
       FROM SPFLI INTO CORRESPONDING FIELDS OF WA
       WHERE CARRID = 'LH' AND
             CONNID = '0400'.

WRITE WA-CARRID.
WRITE WA-CONNID.
WRITE WA-COUNTRYFR.

*New Syntax
REPORT Z.

DATA WA TYPE SPFLI.

SELECT SINGLE *
       FROM SPFLI
       WHERE CARRID = 'LH' AND
             CONNID = '0400'
       INTO CORRESPONDING FIELDS OF @WA.

WRITE WA-CARRID.
WRITE WA-CONNID.
WRITE WA-COUNTRYFR. 

WRITE WA-CARRID.
WRITE WA-CONNID.
WRITE WA-COUNTRYFR.
