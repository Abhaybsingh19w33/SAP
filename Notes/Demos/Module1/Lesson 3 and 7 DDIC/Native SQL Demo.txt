REPORT Z.

DATA: BEGIN OF WA,
        CITYFROM TYPE SPFLI-CITYFROM,
        CITYTO   TYPE SPFLI-CITYTO,
      END OF WA.

EXEC SQL.
  SELECT CITYFROM,  CITYTO
         INTO :WA
         FROM SPFLI
         WHERE MANDT  = '100' AND
               CARRID = 'LH'  AND CONNID = '0400'
ENDEXEC.

WRITE WA-CITYFROM.
WRITE WA-CITYTO.
