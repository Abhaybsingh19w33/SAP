REPORT  Z.
TYPES : BEGIN OF TY_SCARR,
          MANDT    TYPE SCARR-MANDT,
          CARRID   TYPE SCARR-CARRID,
          CARRNAME TYPE SCARR-CARRNAME,
          CURRCODE TYPE SCARR-CURRCODE,
          URL      TYPE SCARR-URL,
        END OF TY_SCARR.

DATA ITSCARR TYPE TABLE OF TY_SCARR INITIAL SIZE 0.
DATA WA TYPE TY_SCARR.

PERFORM SELECTFROMDB.
PERFORM INSERT_INTO_INTTABLE.
PERFORM DISP.

FORM SELECTFROMDB.
  SELECT MANDT CARRID CARRNAME CURRCODE URL  UP TO 5 ROWS
      FROM SCARR
      INTO TABLE ITSCARR.
ENDFORM.


FORM INSERT_INTO_INTTABLE.
WA-MANDT = SY-MANDT. WA-CARRID = 'AI'. WA-CARRNAME  = 'Air India'.
WA-CURRCODE = 'INR'. WA-URL = 'www.Airindia.com'.

*1.
*INSERT WA INTO TABLE ITSCARR.
* Record is APPENDED AT LAST.
*NO INDEX CAN BE SPECIFIED HERE

*2
*INSERT WA INTO ITSCARR INDEX 2.
* record is inserted at index 2

*3
*INSERT INITIAL LINE INTO TABLE ITSCARR.
*if u specify insert inital line
*a blank line is inserted at last.
*NO INDEX CAN BE SPECIFIED HERE

*4
INSERT INITIAL LINE INTO ITSCARR INDEX 2.

ENDFORM.

FORM DISP.
  LOOP AT ITSCARR INTO WA.
    WRITE:/ 'MANDT     :', WA-MANDT.
    WRITE:  'CARRID IS :', WA-CARRID.
    WRITE:  'CARRNAME IS :', WA-CARRNAME.
    WRITE:  'CURRCODE IS :', WA-CURRCODE.
    WRITE:  'URL      IS :', WA-URL.
  ENDLOOP.
ENDFORM.
