REPORT Z NO STANDARD PAGE HEADING.
TYPES : BEGIN OF TY_SCARR,
          MANDT    TYPE SCARR-MANDT,
          CARRID   TYPE SCARR-CARRID,
          CARRNAME TYPE SCARR-CARRNAME,
          CURRCODE TYPE SCARR-CURRCODE,
          URL      TYPE SCARR-URL,
        END OF TY_SCARR.

DATA ITSCARR TYPE TABLE OF TY_SCARR
      WITH NON-UNIQUE DEFAULT KEY
      INITIAL SIZE 0.
DATA WA TYPE TY_SCARR.

PERFORM SELECTFROMDB.
PERFORM APPEND_TO_INTTABLE.

PERFORM DISP.

FORM SELECTFROMDB.
  SELECT MANDT CARRID CARRNAME CURRCODE URL
 FROM SCARR UP TO 5 ROWS INTO TABLE ITSCARR.
ENDFORM.


FORM APPEND_TO_INTTABLE.
  APPEND INITIAL LINE TO ITSCARR.
*  IF U SPECIFY APPEND INITAL LINE A BLANK LINE IS INSERTED AT LAST
ENDFORM.

FORM DISP.
  LOOP AT ITSCARR INTO WA.
   WRITE:/ 'CARRID IS :', WA-CARRID.
   WRITE: 'CARRNAME IS :', WA-CARRNAME.
   WRITE: 'CURRCODE IS :', WA-CURRCODE.
  ENDLOOP.
ENDFORM.
