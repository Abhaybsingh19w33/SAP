REPORT Z.
DATA F1(5) TYPE C VALUE 'ABcde'.
DATA F2(3) TYPE C VALUE '*b*'.
DATA F3(4) TYPE C VALUE '*b#*'.

*PERFORM CP1.
PERFORM CP2.


FORM CP1.
* operator: CP
  IF F1 CP F2.              "True
    WRITE 'True'.
  ELSE.
    WRITE 'False'.
  ENDIF.
  WRITE: /  'sy-fdpos = ', SY-FDPOS.

ENDFORM.


FORM CP2.
* operator: CP
  IF F1 CP F3.              "True
    WRITE 'True'.
  ELSE.
    WRITE 'False'.
  ENDIF.
  WRITE: /  'sy-fdpos = ', SY-FDPOS.

ENDFORM.
