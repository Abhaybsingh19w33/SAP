REPORT Z.
DATA F1(3) TYPE C VALUE 'BD '.
DATA F2(5) TYPE C VALUE 'ABCD '.
DATA F3(5) TYPE C VALUE 'ABCDE'.

*PERFORM CO1.
PERFORM CO2.

FORM CO1.
* operator: CO
*True, if operand1 only contains characters from operand2
  IF F1 CO F2.              "True
    WRITE 'True'.
  ELSE.
    WRITE 'False'.
  ENDIF.
  WRITE: /  'sy-fdpos = ', SY-FDPOS.

ENDFORM.

FORM CO2.
* operator: CO
*True, if operand1 only contains characters from operand2
  IF F1 CO F3.              "True
    WRITE 'True'.
  ELSE.
    WRITE 'False'.
  ENDIF.
  WRITE: /  'sy-fdpos = ', SY-FDPOS.

ENDFORM.
