REPORT Z.
PARAMETERS: NUM1 TYPE  I,
            NUM2 TYPE  I.

AT SELECTION-SCREEN ON NUM1.
  PERFORM VALIDATENUM1.

AT SELECTION-SCREEN ON NUM2.
  PERFORM VALIDATENUM2.

START-OF-SELECTION.
  WRITE  NUM1.
  WRITE NUM2.

FORM VALIDATENUM1.
  IF NUM1 < 0.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Num1 cannot be -ve'.
  ENDIF.
ENDFORM.


FORM VALIDATENUM2.
  IF NUM2 < 10.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Num2 cannot be less than 10'.
  ENDIF.
ENDFORM.
