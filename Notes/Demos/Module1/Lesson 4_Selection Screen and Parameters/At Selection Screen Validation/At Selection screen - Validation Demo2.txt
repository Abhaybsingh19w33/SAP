REPORT Z.
PARAMETERS: NUM1 TYPE  I,
            NUM2 TYPE  I.

AT SELECTION-SCREEN.
  PERFORM VALIDATENUM1NUM2.
  
START-OF-SELECTION.
  WRITE NUM1.
  WRITE NUM2.

FORM VALIDATENUM1NUM2.
  IF NUM1 < 0.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Num1 cannot be -ve'.
  ENDIF.
  IF NUM2 < 10.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Num2 cannot be less than 10'.
  ENDIF.
ENDFORM.
