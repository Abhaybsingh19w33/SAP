REPORT Z.
PARAMETERS: NUM TYPE  I,
            NAME(10) TYPE  C.

AT SELECTION-SCREEN ON NUM.
  PERFORM VALIDATENUM.

AT SELECTION-SCREEN ON NAME.
  PERFORM VALIDATENAME.

START-OF-SELECTION.
  WRITE:'Num  is:', NUM.
  WRITE:'Name is:', NAME.

FORM VALIDATENUM.
  IF NUM IS INITIAL.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Num cannot be 0'.
  ENDIF.
ENDFORM.


FORM VALIDATENAME.
  IF NAME IS INITIAL.
    MESSAGE E001(Z104329MSGCLS)
    WITH 'Name cannot be initial'.
  ENDIF.
ENDFORM.
