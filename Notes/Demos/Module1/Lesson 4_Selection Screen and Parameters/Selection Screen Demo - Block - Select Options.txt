REPORT Z.
TYPES : BEGIN OF TYSCARR,
          CARRID   TYPE S_CARR_ID,
          CARRNAME TYPE S_CARRNAME,
          CURRCODE TYPE S_CURRCODE,
        END OF TYSCARR.
DATA ITAB TYPE TABLE OF TYSCARR INITIAL SIZE 0.
DATA WA TYPE TYSCARR.
DATA: V_CARRID TYPE S_CARR_ID,
      V_CURR   TYPE S_CURRCODE.

SELECTION-SCREEN BEGIN OF BLOCK B1 WITH FRAME TITLE TEXT-001.
SELECT-OPTIONS S_CARRID FOR V_CARRID.
SELECTION-SCREEN END OF BLOCK B1.

SELECTION-SCREEN BEGIN OF BLOCK B2 WITH FRAME TITLE TEXT-002.
PARAMETERS : RB1 RADIOBUTTON GROUP G1,
             RB2 RADIOBUTTON GROUP G1.
SELECTION-SCREEN END OF BLOCK B2.

START-OF-SELECTION.
  IF RB1 = 'X'.
    V_CURR = 'USD'.
  ELSEIF RB2 = 'X'.
    V_CURR = 'EUR'.
  ENDIF.

  SELECT CARRID CARRNAME CURRCODE INTO TABLE ITAB
  FROM SCARR  UP TO 10 ROWS WHERE
  CARRID IN S_CARRID AND CURRCODE EQ V_CURR.
  LOOP AT ITAB INTO WA.
    WRITE :/ WA-CARRID, WA-CARRNAME, WA-CURRCODE.
  ENDLOOP.
