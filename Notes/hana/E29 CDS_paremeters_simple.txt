define view yt_CDS_para
       with parameters sales_doc : abap.char( 10 ),
                         percent : abap.int1
as select from zrt_sales {
    kunnr,
    vbeln,
    netpr,
    :percent as discount_percent,
    netpr * DIVISION( :percent , 100 ,2 ) as percentage
 
} where vbeln = :sales_doc
